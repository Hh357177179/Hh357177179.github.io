# 整体规范

注释名 内容 含义
@description 文件、方法描述 描述该功能模块的作用
@author 作者信息 描述此函数作者的信息
@version 版本号 描述版本信息
@param 参数名 {类型} 描述 描述此函数入参的信息
@return 返回值 {类型} 描述 描述此函数返回的信息
  • 通用的.css, .js：@description @author @version
  • 项目下公共.js：@description
  • 公共方法：@description @param @return

# 块级注释

块级注释，包括对文件的整体注释，以及方法块注释

例：/src/utils/common/validate.js

/**
 * @description 正则校验方法
 * @author zhanghaizhi
 */

const num = /^[0-9]*$/
const mobile = /^1[3|4|5|7|8|9][0-9]\d{8}$/
const email = /^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/
const emoji = /\ud83c[\udf00-\udfff]|\ud83d[\udc00-\ude4f]|\ud83d[\ude80-\udeff]|[\uD800-\uDBFF][\uDC00-\uDFFF]/

const validate = {
/**
 * @description 数字检验
 * @param {String} str 校验字符串
 * @return {Boolean} 校验结果
 */
  isNumber (str) {
    return num.test(str)
  },

/**
 * @description 手机号校验
 * @param {String} str 校验字符串
 * @return {Boolean} 校验结果
 */
  isMobile (str) {
    return mobile.test(str)
  },

/**
 * @description 邮箱校验
 * @param {String} str 校验字符串
 * @return {Boolean} 校验结果
 */
  isEmail (str) {
    return email.test(str)
  },

/**
 * @description 表情符号校验
 * @param {String} str 校验字符串
 * @return {Boolean} 校验结果
 */
  hasEmoji (str) {
    return emoji.test(str)
  }
}

export default validate

# 全局变量注释

vuex 数据状态管理的变量必须注释

// store/state.js
const state = {
  appId: 'wx01d071fbf6dac44c', // 应用的唯一标识
  openId: null, // 一个用户打开一个微信应用（公众号、小程序）的唯一标识
  unionId: null, // 同一用户，对同一个微信开放平台下的不同应用，unionid是相同的
}

# 页面变量注释

单个.vue页面，基础的、全局的、路由传进来的变量必须注释，例：type status 等

定义规则：是否关系用0,1；平级关系用1,2,3...；默认值为Number类型

data () {
  return {
    type: 1, // 1灯控 2空调 3水表
    status: 0, // 0关闭 1开启
  }
}

# 逻辑注释

方法中复杂易混淆的逻辑必须注释

如果在书写代码时，业务逻辑过于复杂，需要在特定的地方加上辅助注释信息，以备后来开发人员理解业务
